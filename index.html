let audioCtx=null,playing=false,musicLoop=null;

function note(f,t,d,v=0.03,type='square'){

 const o=audioCtx.createOscillator();

 const g=audioCtx.createGain();

 o.type=type;o.frequency.value=f;

 g.gain.setValueAtTime(v,t);

 g.gain.exponentialRampToValueAtTime(0.0001,t+d);

 o.connect(g);g.connect(audioCtx.destination);

 o.start(t);o.stop(t+d);

}

function startMusic(){

 audioCtx=new (window.AudioContext||window.webkitAudioContext)();

 const melody=[523,659,784,880,784,659,698,784];

 const bass=[131,147,165,175,165,147,131,110];

 let s=0;

 musicLoop=setInterval(()=>{

   const n=audioCtx.currentTime;

   note(melody[s%8],n,0.28,.035,'square');

   note(melody[s%8]*1.25,n,0.25,.015,'square');

   note(bass[s%8],n,0.45,.025,'triangle');

   if(s%2===0){

      note(melody[(s+2)%8]/2,n,0.35,.012,'sawtooth');

   }

   s++;

 },300);

}

function toggleMusic(){

 if(!playing){startMusic();playing=true;}

 else{

   clearInterval(musicLoop);

   audioCtx.close();

   playing=false;

 }

}
